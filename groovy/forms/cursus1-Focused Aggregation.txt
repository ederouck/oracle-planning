Set<String> editedEntities = [] // declare a set of members, needs to be unique
// for each edited cell, store the name of the member of the entity into the editedEntities 
operation.grid.dataCellIterator({DataCell cell ->
	cell.edited}).each { DataCell cell ->
    	editedEntities << cell.getMemberName("Entity")
}

// Make a list of Pov members as a string
List<String> povMemberNames =
// store all the essbase members for this
operation.grid.pov*.essbaseMbrName

// Start building up the variable calcscript
String calcScript ="""
    FIX("${povMemberNames.join('", "')}",
    "${editedEntities.join('", "')}"
    )
	CALC Dim("Product");
    ENDFIX
    """
    
println("The following calc script was executed by $operation.user.fullName: \n $calcScript")

return calcScript
