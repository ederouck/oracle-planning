/* RTPS: */
if (!operation.hasGrid()) {
    throwVetoException("No grid available to process!")
}

def grid = operation.grid

grid.dataCellIterator().each { cell ->
    if (cell.data == 1.0) {
        cell.bgColor = 0x3cb371 // Green for Open
    } else if (cell.data == 2.0) {
        cell.bgColor = 0x4169e1 // Blue for Locked
    }
}
